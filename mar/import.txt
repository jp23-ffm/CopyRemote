  - import_config.py — la table de mapping + les transformers (Python pur, pas YAML : les transformers sont déjà des

  - import_csv.py — le script d'import générique qui consomme la config

  import_config.py

  from datetime import datetime

  # ── Transformers réutilisables ─────────────────────────────────────────

  def normalize_date(val):
      """Essaie plusieurs formats de date."""
      if not val or not val.strip():
          return None
      for fmt in ('%Y-%m-%d', '%d/%m/%Y', '%m/%d/%Y', '%d-%m-%Y', '%Y%m%d'):
          try:
              return datetime.strptime(val.strip(), fmt).date()
          except ValueError:
              pass
      return None  # ou raise si on veut être strict

  def upper_strip(val):
      return val.strip().upper() if val else ''

  def title_strip(val):
      return val.strip().title() if val else ''


  # ── Mapping champ CSV → champ modèle ──────────────────────────────────
  # Format : "Nom exact dans le CSV" : ("champ_modèle", transformer_ou_None)
  # Les champs CSV non listés ici sont ignorés.

  FIELD_MAP = {
      "Server Name":          ("SERVER_ID",    str.strip),
      "OS Family ":           ("OSFAMILY",     upper_strip),   # espace dans le header CSV
      "OS Short Name":        ("OSSHORTNAME",  str.strip),
      "Machine Type":         ("MACHINE_TYPE", upper_strip),
      "Region":               ("REGION",       upper_strip),
      "Country":              ("COUNTRY",      title_strip),
      "Last Analysis Date":   ("analysis_date", normalize_date),
      "Live Status":          ("LIVE_STATUS",   upper_strip),
      "Snow Status":          ("SNOW_STATUS",   upper_strip),
      "Support Group":        ("SNOW_SUPPORTGROUP", str.strip),
  }

  # Champs à lire du CSV mais utiles seulement dans les row transforms
  EXTRA_READ = {"Alt OS Name", "Secondary Status", "Date Created", "Date Modified"}


  # ── Row-level transforms ───────────────────────────────────────────────
  # Reçoit le dict déjà mappé + la ligne brute du CSV.
  # Peut modifier/compléter le dict mappé et retourne le dict final.

  def row_transform(mapped: dict, raw: dict) -> dict:
      # Exemple : préférer "Alt OS Name" si OSFAMILY est vide
      if not mapped.get('OSFAMILY') and raw.get('Alt OS Name', '').strip():
          mapped['OSFAMILY'] = raw['Alt OS Name'].strip().upper()

      # Exemple : prendre la date la plus récente entre deux colonnes
      d1 = normalize_date(raw.get('Date Created', ''))
      d2 = normalize_date(raw.get('Date Modified', ''))
      if d1 and d2:
          mapped['analysis_date'] = max(d1, d2)
      elif d1 or d2:
          mapped['analysis_date'] = d1 or d2

      # Exemple : dériver un champ calculé
      if mapped.get('LIVE_STATUS') == 'ALIVE' and mapped.get('SNOW_STATUS') == 'OPERATIONAL':
          mapped['alive_status_inconsistent'] = 'OK'

      return mapped

  import_csv.py

  import csv
  import logging
  from pathlib import Path
  from django.db import transaction
  from .models import ServerDiscrepancy
  from .import_config import FIELD_MAP, EXTRA_READ, row_transform

  log = logging.getLogger(__name__)
  BATCH_SIZE = 500

  # Champs du modèle qu'on met à jour si le serveur existe déjà (upsert)
  UPDATE_FIELDS = [model_field for model_field, _ in FIELD_MAP.values()
                   if model_field != 'SERVER_ID']


  def _normalize_headers(headers: list[str]) -> dict[str, str]:
      """Retourne un mapping header_original → header_stripped pour la recherche."""
      return {h: h.strip() for h in headers}


  def import_csv(filepath: str | Path) -> dict:
      path = Path(filepath)
      to_create, errors = [], []

      with open(path, newline='', encoding='utf-8-sig') as f:
          reader = csv.DictReader(f)

          # Normaliser les headers une fois (strip des espaces)
          raw_headers = reader.fieldnames or []
          header_map = {h.strip(): h for h in raw_headers}  # stripped → original

          # Vérifier que les colonnes requises sont présentes
          missing = [col for col in FIELD_MAP if col.strip() not in header_map]
          if missing:
              log.warning("Colonnes absentes du CSV : %s", missing)

          for lineno, raw_row in enumerate(reader, start=2):
              # Re-indexer la ligne par header strippé
              row = {h.strip(): v for h, v in raw_row.items()}

              try:
                  mapped = {}
                  for csv_col, (model_field, transformer) in FIELD_MAP.items():
                      value = row.get(csv_col.strip(), '') or ''
                      mapped[model_field] = transformer(value) if transformer else value

                  # Lire les champs extra pour les row transforms
                  extra = {col: row.get(col, '') for col in EXTRA_READ}
                  mapped = row_transform(mapped, {**row, **extra})

                  # Ignorer les lignes sans SERVER_ID
                  if not mapped.get('SERVER_ID'):
                      errors.append(f"Ligne {lineno} : SERVER_ID vide, ignorée")
                      continue

                  to_create.append(ServerDiscrepancy(**mapped))

              except Exception as e:
                  errors.append(f"Ligne {lineno} : {e}")
                  log.debug("Ligne %d erreur", lineno, exc_info=True)

      # ── Bulk upsert ────────────────────────────────────────────────────
      inserted = updated = 0
      if to_create:
          with transaction.atomic():
              result = ServerDiscrepancy.objects.bulk_create(
                  to_create,
                  batch_size=BATCH_SIZE,
                  update_conflicts=True,          # upsert si SERVER_ID existe déjà
                  unique_fields=['SERVER_ID'],
                  update_fields=UPDATE_FIELDS,
              )
          inserted = sum(1 for o in result if o._state.adding)
          updated  = len(result) - inserted

      summary = {
          'total':    len(to_create) + len(errors),
          'inserted': inserted,
          'updated':  updated,
          'errors':   errors,
      }
      log.info("Import terminé : %s", summary)
      return summary

  Points clés

  ┌─────────────────────────────┬───────────────────────────────────────────────────────────────────────────────────┐
  │            Sujet            │                                       Choix                                       │
  ├─────────────────────────────┼───────────────────────────────────────────────────────────────────────────────────┤
  │ Mapping dans                │ import_config.py (Python) — les transformers y sont naturels                      │
  ├─────────────────────────────┼───────────────────────────────────────────────────────────────────────────────────┤
  │ Headers CSV sales           │ Strip systématique à la lecture, matching sur la version strippée                 │
  ├─────────────────────────────┼───────────────────────────────────────────────────────────────────────────────────┤
  │ Transformers                │ 2 niveaux : field-level (1 valeur → 1 valeur) et row-level (accès à toute la      │
  │                             │ ligne)                                                                            │
  ├─────────────────────────────┼───────────────────────────────────────────────────────────────────────────────────┤
  │ Bulk                        │ bulk_create avec update_conflicts=True (upsert) — pas de boucle save()            │
  ├─────────────────────────────┼───────────────────────────────────────────────────────────────────────────────────┤
  │ Champs lus mais non         │ EXTRA_READ : lus pour les row transforms, pas dans le modèle                      │
  │ injectés                    │                                                                                   │
  ├─────────────────────────────┼───────────────────────────────────────────────────────────────────────────────────┤
  │ Erreurs                     │ Collectées ligne par ligne, n'arrêtent pas le traitement                          │
  └─────────────────────────────┴───────────────────────────────────────────────────────────────────────────────────┘

  Pour ajouter un nouveau champ à importer : une ligne dans FIELD_MAP. Pour une transformation complexe (cross-field) :
  dans row_transform. La logique métier reste séparée du mécanisme d'import.


-------------------------------------------------------------------------------------------------------------------------------

Upsert = "insert ou update si la clé existe déjà". Ici ça voulait dire : si un SERVER_ID est déjà en base, on écrase
  ses valeurs plutôt que d'insérer un doublon.

  Si les doublons sont OK, on simplifie bulk_create — plus besoin de tout ça :

  # Avant (upsert)
  ServerDiscrepancy.objects.bulk_create(
      to_create,
      batch_size=BATCH_SIZE,
      update_conflicts=True,
      unique_fields=['SERVER_ID'],
      update_fields=UPDATE_FIELDS,
  )

  # Après (insert simple)
  ServerDiscrepancy.objects.bulk_create(
      to_create,
      batch_size=BATCH_SIZE,
  )

  Et on peut supprimer UPDATE_FIELDS du fichier, qui ne sert plus à rien.
