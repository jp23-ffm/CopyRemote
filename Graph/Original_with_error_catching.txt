Function R_CheckReplyMail {
    param(
        [Parameter(Mandatory = $true)] [string]$S_Mail,
        [Parameter(Mandatory = $true)] [string]$S_LoginExchange,
        [Parameter(Mandatory = $true)] [string]$S_PasswordExchange,
        [Parameter(Mandatory = $true)] [string]$S_Domain,
        [Parameter(Mandatory = $true)] [string]$S_FolderRoot,
        [Parameter(Mandatory = $true)] [string]$S_FolderProvider,
        [Parameter(Mandatory = $true)] [string]$S_FolderSource,
        [Parameter(Mandatory = $true)] [string]$S_FolderDestination,
        [Parameter(Mandatory = $true)] [string]$S_FolderError,
        [Parameter(Mandatory = $true)] [string]$S_Connectionstring,
        [Parameter(Mandatory = $true)] [string]$S_Css
    )
    
    $I_adOpenStatic = 3
    $I_adLockOptimistic = 3
    $O_Connection = New-Object -comobject ADODB.Connection
    $O_Connection.Open($S_Connectionstring)
    $O_RecordsetInsertUpdate = New-Object -comobject ADODB.Recordset
    $O_RecordsetRead = New-Object -comobject ADODB.Recordset
    $O_RecordsetReadSecond = New-Object -comobject ADODB.Recordset
    
    # === POINT CRITIQUE 1 : Initialisation Graph ===
    Try {
        $connectionResult = Initialize-EwsCompatConnection -UserEmail $S_Mail
        if (-not $connectionResult.Success) {
            throw "Graph connection failed"
        }
    }
    Catch {
        Write-host -f red "ERROR Graph connection: $_"
        R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReplyMail] - Graph connection failed"
        $O_Connection.Close()
        Return
    }
    
    # === POINT CRITIQUE 2 : Navigation dossiers ===
    Try {
        $O_FolderRootMailAutomate = Get-EwsCompatFolder -ParentFolderId "Inbox" -DisplayName $S_FolderRoot
    }
    Catch {
        Write-host -f red "ERROR Graph folder navigation: $_"
        R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReplyMail] - Cannot access folders"
        $O_Connection.Close()
        Return
    }
    
    If ($O_FolderRootMailAutomate -eq $Null) {
        Write-host -f green "Folder Root not found"
        R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReplyMail], Folder $S_FolderRoot not available"
    }
    Else {
        Try {
            $O_FolderProviderMailAutomate = Get-EwsCompatFolder -ParentFolderId $O_FolderRootMailAutomate[0].Id -DisplayName $S_FolderProvider
        }
        Catch {
            Write-host -f red "ERROR Graph folder navigation: $_"
            R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReplyMail] - Cannot access provider folder"
            $O_Connection.Close()
            Return
        }
        
        If ($O_FolderProviderMailAutomate -eq $Null) {
            Write-host -f green "Folder Provider not found"
            R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReplyMail], Folder $S_FolderProvider not available"
        }
        Else {
            Try {
                $allSubFolders = Get-EwsCompatFolder -ParentFolderId $O_FolderProviderMailAutomate[0].Id
            }
            Catch {
                Write-host -f red "ERROR Graph folder navigation: $_"
                R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReplyMail] - Cannot access subfolders"
                $O_Connection.Close()
                Return
            }
            
            $O_FolderTODOMailAutomate = $allSubFolders | ?{$_.DisplayName -eq $S_FolderSource}
            $O_FolderDONEMailAutomate = $allSubFolders | ?{$_.DisplayName -eq $S_FolderDestination}
            $O_FolderERRORMailAutomate = $allSubFolders | ?{$_.DisplayName -eq $S_FolderError}
            
            $I_CountMailchecked = 0
            
            If ($O_FolderTODOMailAutomate -eq $Null) {
                Write-host -f green "folder TODO not found"
                R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReplyMail], Folder TODO not available"
            }
            Else {
                If ($O_FolderDONEMailAutomate -eq $Null) {
                    Write-host -f green "Folder DONE not found"
                    R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReplyMail], Folder DONE not available"
                }
                Else {
                    If ($O_FolderERRORMailAutomate -eq $Null) {
                        Write-host -f green "Folder ERROR not found"
                        R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReplyMail], Folder ERROR not available"
                    }
                    Else {
                        Write-host -f green "Requete dans une requete à faire"
                        
                        $O_RecordsetRead.Open(
                            "select T_PurchaseCart_ID From T_PurchaseCart where T_PurchaseCart_StateInt = 13", 
                            $O_Connection, 
                            $I_adOpenStatic, 
                            $I_adLockOptimistic
                        )
                        
                        If (-Not $O_RecordsetRead.EOF) {
                            $O_RecordsetRead.MoveFirst()
                            
                            Do {
                                # === POINT CRITIQUE 3 : Recherche et réponse ===
                                Try {
                                    Write-host -f green "envoie mail"
                                    
                                    $O_RecordsetReadSecond.Open(
                                        "select top 1 T_PurchaseMailReceived_ReplyMail, T_PurchaseMailReceived_T_PurchaseCart_ID " +
                                        "From T_PurchaseMailReceived " +
                                        "Where T_PurchaseMailReceived_T_PurchaseCart_ID = " + 
                                        $($O_RecordsetRead.Fields.Item("T_PurchaseCart_ID").Value) + 
                                        " order by T_PurchaseMailReceived_ID desc", 
                                        $O_Connection, 
                                        $I_adOpenStatic, 
                                        $I_adLockOptimistic
                                    )
                                    
                                    $O_RecordsetReadSecond.MoveFirst()
                                    $I_T_PurchaseMailReceived_T_PurchaseCart_ID = $($O_RecordsetReadSecond.Fields.Item("T_PurchaseMailReceived_T_PurchaseCart_ID").Value)
                                    
                                    $searchSubject = "[PURCHASE_WORKFLOW_ORDER][ID_$I_T_PurchaseMailReceived_T_PurchaseCart_ID]"
                                    
                                    $allMessages = Get-MgUserMailFolderMessage -UserId $script:EwsCompat_CurrentUser `
                                                                               -MailFolderId $O_FolderDONEMailAutomate.Id `
                                                                               -Filter "contains(subject, '$searchSubject')" `
                                                                               -Top 1
                                    
                                    if ($allMessages) {
                                        $O_ReplyOnMail = New-EwsCompatMessage -GraphMessage $allMessages[0] -GraphAttachments $null
                                        
                                        $replyBody = $O_RecordsetReadSecond.Fields.Item("T_PurchaseMailReceived_ReplyMail").Value
                                        
                                        $replyResult = Send-EwsCompatReply -MessageId $O_ReplyOnMail.Id `
                                                                           -ReplyBody $replyBody `
                                                                           -ReplyToAll $true
                                        
                                        $O_RecordsetInsertUpdate.Open(
                                            "Update T_PurchaseCart set T_PurchaseCart_StateInt = '14', T_PurchaseCart_StateString = 'REPLY SENT' " +
                                            "where T_PurchaseCart_ID = '" + $O_RecordsetRead.Fields.Item("T_PurchaseCart_ID").Value + "'", 
                                            $O_Connection, 
                                            $I_adOpenStatic, 
                                            $I_adLockOptimistic
                                        )
                                    }
                                    
                                    $O_RecordsetReadSecond.Close()
                                }
                                Catch {
                                    Write-host -f red "ERROR Graph reply: $_"
                                    R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReplyMail] - Cannot send reply"
                                    Try { $O_RecordsetReadSecond.Close() } Catch {}
                                    Break
                                }
                                
                                $O_RecordsetRead.MoveNext()
                                
                            } Until ($O_RecordsetRead.EOF)
                        }
                        Else {
                            Write-host -f green "None Mail to do it"
                        }
                    }
                }
            }
        }
    }
    
    $O_RecordsetRead.Close()
    $O_Connection.Close()
}

Function R_CheckReceivedMail {
    param(
        [Parameter(Mandatory = $true)] [string]$S_Mail,
        [Parameter(Mandatory = $true)] [string]$S_LoginExchange,
        [Parameter(Mandatory = $true)] [string]$S_PasswordExchange,
        [Parameter(Mandatory = $true)] [string]$S_Domain,
        [Parameter(Mandatory = $true)] [string]$S_FolderRoot,
        [Parameter(Mandatory = $true)] [string]$S_FolderProvider,
        [Parameter(Mandatory = $true)] [string]$S_FolderSource,
        [Parameter(Mandatory = $true)] [string]$S_FolderDestination,
        [Parameter(Mandatory = $true)] [string]$S_FolderError,
        [Parameter(Mandatory = $true)] [string]$S_PathArchiveAttachments,
        [Parameter(Mandatory = $true)] [string]$S_Connectionstring
    )
    
    $I_adOpenStatic = 3
    $I_adLockOptimistic = 3
    $O_Connection = New-Object -comobject ADODB.Connection
    $O_Connection.Open($S_Connectionstring)
    $O_RecordsetInsertUpdate = New-Object -comobject ADODB.Recordset
    
    # === POINT CRITIQUE 1 : Initialisation Graph ===
    Try {
        $connectionResult = Initialize-EwsCompatConnection -UserEmail $S_Mail
        if (-not $connectionResult.Success) {
            throw "Graph connection failed"
        }
    }
    Catch {
        Write-host -f red "ERROR Graph connection: $_"
        R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReceivedMail] - Graph connection failed"
        $O_Connection.Close()
        Return
    }
    
    # === POINT CRITIQUE 2 : Navigation dossiers ===
    Try {
        $O_FolderRootMailAutomate = Get-EwsCompatFolder -ParentFolderId "Inbox" -DisplayName $S_FolderRoot
    }
    Catch {
        Write-host -f red "ERROR Graph folder navigation: $_"
        R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReceivedMail] - Cannot access folders"
        $O_Connection.Close()
        Return
    }
    
    If ($O_FolderRootMailAutomate -eq $Null) {
        Write-host -f green "Folder Provider not found"
        R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReceivedMail], Folder $S_FolderRoot not available"
    }
    Else {
        Try {
            $O_FolderProviderMailAutomate = Get-EwsCompatFolder -ParentFolderId $O_FolderRootMailAutomate[0].Id -DisplayName $S_FolderProvider
        }
        Catch {
            Write-host -f red "ERROR Graph folder navigation: $_"
            R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReceivedMail] - Cannot access provider folder"
            $O_Connection.Close()
            Return
        }
        
        If ($O_FolderProviderMailAutomate -eq $Null) {
            Write-host -f green "Folder Provider not found"
            R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReceivedMail], Folder $S_FolderProvider not available"
        }
        Else {
            Try {
                $allSubFolders = Get-EwsCompatFolder -ParentFolderId $O_FolderProviderMailAutomate[0].Id
            }
            Catch {
                Write-host -f red "ERROR Graph folder navigation: $_"
                R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReceivedMail] - Cannot access subfolders"
                $O_Connection.Close()
                Return
            }
            
            $O_FolderTODOMailAutomate = $allSubFolders | ?{$_.DisplayName -eq $S_FolderSource}
            $O_FolderDONEMailAutomate = $allSubFolders | ?{$_.DisplayName -eq $S_FolderDestination}
            $O_FolderERRORMailAutomate = $allSubFolders | ?{$_.DisplayName -eq $S_FolderError}
            
            $I_CountMailchecked = 0
            
            If ($O_FolderTODOMailAutomate -eq $Null) {
                Write-host -f green "folder TODO not found"
                R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReceivedMail], Folder TODO not available"
            }
            Else {
                If ($O_FolderDONEMailAutomate -eq $Null) {
                    Write-host -f green "Folder DONE not found"
                    R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReceivedMail], Folder DONE not available"
                }
                Else {
                    If ($O_FolderERRORMailAutomate -eq $Null) {
                        Write-host -f green "Folder ERROR not found"
                        R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReceivedMail], Folder ERROR not available"
                    }
                    Else {
                        If ($O_FolderTODOMailAutomate.TotalCount -ne 0) {
                            $I_CountMail = $O_FolderTODOMailAutomate.TotalCount
                            Write-host -f green "number mail to checking $I_CountMail"
                            
                            Do {
                                # === POINT CRITIQUE 3 : Récupération messages ===
                                Try {
                                    $O_CheckMailCollection = Get-EwsCompatMessages -FolderId $O_FolderTODOMailAutomate.Id -Top 1 -OrderBy "DateTimeReceived" -Ascending $true
                                    
                                    $O_CheckMailCollection.Load()
                                    $O_CheckMail = $O_CheckMailCollection.Items[0]
                                    
                                    Try {
                                        $O_CheckMail.Attachments | ForEach-Object { $_.Load() }
                                    }
                                    Catch {
                                        Write-host -f green "error onload attchment"
                                    }
                                }
                                Catch {
                                    Write-host -f red "ERROR Graph get messages: $_"
                                    R_Send-MailSmtp -Subject "ERROR processing WorkFlow Order [R_CheckReceivedMail] - Cannot retrieve messages"
                                    Break
                                }
                                
                                If ($O_CheckMail.Subject -like '*[PURCHASE WORKFLOW ORDER]*' -And $O_CheckMail.Subject -like "*ID_*") {
                                    Write-host -f green "Format de mail conforme a traiter"
                                    Write-host -f green $($O_CheckMail.Subject)
                                    Write-host -f green $($O_CheckMail.From.Address)

                                        $O_CheckMail.items.Move($O_FolderERRORMailAutomate.Id)
                                    }
                                    Catch {
                                        Write-host -f red "ERROR moving message: $_"
                                    }
                                }
                                
                                $I_CountMailchecked++
                                Start-Sleep -s 10
                                
                            } Until ($I_CountMail -eq $I_CountMailchecked)
                        }
                        Else {
                            Write-host -f green "None Mail to do it"
                        }
                    }
                }
            }
        }
    }
    
    $O_Connection.Close()
}
 

Function F_SentMail {
    param(
        [Parameter(Mandatory = $true)] [string]$S_MailOfService,
        [Parameter(Mandatory = $true)] [string]$S_LoginExchange,
        [Parameter(Mandatory = $true)] [string]$S_Domain,
        [Parameter(Mandatory = $true)] [string]$S_PasswordExchange,
        [Parameter(Mandatory = $true)] [string]$S_Connectionstring,
        [Parameter(Mandatory = $true)] [string]$S_Body,
        [Parameter(Mandatory = $true)] [string]$S_SubjectMail,
        [Parameter(Mandatory = $true)] [array]$A_ToRecipients,
        [Parameter(Mandatory = $false)] [string]$SmtpServer = "vip-parapop.fr.net.intra",
        [Parameter(Mandatory = $false)] [bool]$SaveToSentItems = $true
    )
    
    Try {
        if ($SaveToSentItems) {
            Initialize-EwsCompatConnection -UserEmail $S_MailOfService
        }
        
        $toRecipients = @()
        foreach ($recipient in $A_ToRecipients) {
            if ($recipient -is [string]) {
                $toRecipients += $recipient
            }
            elseif ($recipient.Address) {
                $toRecipients += $recipient.Address
            }
            elseif ($recipient.EmailAddress) {
                $toRecipients += $recipient.EmailAddress
            }
        }
        
        $securePassword = ConvertTo-SecureString -String $S_PasswordExchange -AsPlainText -Force
        $smtpCredential = New-Object System.Management.Automation.PSCredential("$S_Domain$S_LoginExchange", $securePassword)
        
        Send-MailMessage -From $S_MailOfService `
                        -To $toRecipients `
                        -Subject $S_SubjectMail `
                        -Body $S_Body `
                        -BodyAsHtml `
                        -SmtpServer $SmtpServer `
                        -Credential $smtpCredential `
                        -UseSsl `
                        -Encoding UTF8
        
        if ($SaveToSentItems) {
            Try {
                $toRecipientsGraph = @()
                foreach ($recipient in $toRecipients) {
                    $toRecipientsGraph += @{
                        EmailAddress = @{ Address = $recipient }
                    }
                }
                
                $messageBody = @{
                    Subject = $S_SubjectMail
                    Body = @{
                        ContentType = "HTML"
                        Content = $S_Body
                    }
                    ToRecipients = $toRecipientsGraph
                    From = @{
                        EmailAddress = @{ Address = $S_MailOfService }
                    }
                    SentDateTime = (Get-Date).ToUniversalTime().ToString("yyyy-MM-ddTHH:mm:ssZ")
                    IsDraft = $false
                }
                
                New-MgUserMessage -UserId $S_MailOfService `
                                 -MailFolderId "SentItems" `
                                 -BodyParameter $messageBody `
                                 -ErrorAction SilentlyContinue
            }
            Catch {
            }
        }
        
        Return @(0, $null)
    }
    Catch {
        Return @(1, $_.Exception.Message)
    }
}
