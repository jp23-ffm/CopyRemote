# /etc/httpd/conf.d/chimera.conf  (RHEL/CentOS)
# ou /etc/apache2/sites-available/chimera.conf  (Debian/Ubuntu)

<VirtualHost *:443>
    ServerName chimera.cib.echonet

    # SSL — mêmes certs que nginx
    SSLEngine on
    SSLCertificateFile    /apps/chimera/chimeraiaasprd-1750076670147.pem
    SSLCertificateKeyFile /apps/chimera/chimeraiaasprd_02.key
    SSLProtocol           TLSv1.2 TLSv1.3
    SSLCipherSuite        HIGH:!aNULL:!MD5
    SSLHonorCipherOrder   on

    # Proxy vers Gunicorn — équivalent upstream + proxy_pass
    ProxyRequests     Off
    ProxyPreserveHost On

    ProxyPass         /static/ !
    ProxyPass         /media/  !
    ProxyPass         / http://127.0.0.1:8080/
    ProxyPassReverse  / http://127.0.0.1:8080/

    # Headers forwarded — équivalent proxy_set_header
    RequestHeader set X-Forwarded-Proto "https"
    RequestHeader set X-Real-IP         "%{REMOTE_ADDR}s"

    # Timeouts — tu avais proxy_connect_timeout 2s / proxy_read_timeout 30s
    ProxyTimeout 30

    # Fichiers statiques servis directement par Apache
    Alias /static/ /apps/chimera/chimera-unix-ondev/chimeraunix/static/
    Alias /media/  /apps/chimera/chimera-unix-ondev/chimeraunix/media/

    <Directory /apps/chimera/chimera-unix-ondev/chimeraunix/static>
        Require all granted
    </Directory>
    <Directory /apps/chimera/chimera-unix-ondev/chimeraunix/media>
        Require all granted
    </Directory>

    # Health check — équivalent location /healthlb/
    <Location /healthlb/>
        ProxyPass        http://127.0.0.1:8080/api/health/
        ProxyPassReverse http://127.0.0.1:8080/api/health/
        CustomLog        /dev/null combined
    </Location>

    # Logs
    ErrorLog  /var/log/apache2/chimera_error.log
    CustomLog /var/log/apache2/chimera_access.log combined

</VirtualHost>

# Redirect HTTP → HTTPS
<VirtualHost *:80>
    ServerName chimera.cib.echonet
    Redirect permanent / https://chimera.cib.echonet/
</VirtualHost>
